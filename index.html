Object.observe

var o = {};

Object.observe(o, function(changes) {
  console.log(changes);
});

// Add a property
o.foo = 'bar';

// Change record:
[{ name: "foo",
   object: o,
   type: "add" }]
// Update a property:
o.foo = 123;

// Change record
[{ name: "foo",
   object: o,
   oldValue: "bar",
   type: "update" }]
delete o.foo;

// Change record:
[{ name: "foo",
   object: o,
   oldValue: 123,
   type: "delete" }]
Object.freeze(o);

// Change record:
[{ object: o,
   type: "preventExtensions" }]
